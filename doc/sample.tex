from randassign import RandAssign
ra = RandAssign()
ra.solnfmt = '.tex'

from sympy import *
from sympy.statistics import *
\end{sympycode}

\begin{document}

\maketitle

\begin{sympycode}
distr = Uniform(1, 10)
from sympy.abc import x
p = int(distr.random())
eq = Derivative(sin(x)**p, x)
question = r'What is ${0}$?'.format(latex(eq))
answer = r'\fbox{{${0}$}}'.format(latex(eq.doit()))
ra.soln.append(question + ' ' + answer + '\n')
\end{sympycode}

\sympy{question}

\begin{sympycode}
ra.cleanup()
\end{sympycode}
